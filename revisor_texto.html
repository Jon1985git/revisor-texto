    <button onclick="revisarTexto()">Revisar</button>

    <div class="resultado" id="resultado"></div>


    <script>

        function revisarTexto() {

            const texto = document.getElementById('texto').value;

            let resultado = '';


            // Revisión básica de ortografía y gramática

            const erroresComunes = {

                'aver': 'haber',

                'habrir': 'abrir',

                'hay': 'ahí',

                'halla': 'haya',

                'haci': 'así',

                'asia': 'hacia',

                'bienes': 'vienes',

                'valla': 'vaya'

            };


            let textoRevisado = texto;

            for (let error in erroresComunes) {

                const regex = new RegExp('\\b' + error + '\\b', 'gi');

                textoRevisado = textoRevisado.replace(regex, erroresComunes[error]);

            }


            resultado += '<h3>Texto revisado:</h3>';

            resultado += '<p>' + textoRevisado + '</p>';


            // Convertir a tuteo, ustedeo y voseo

            const tuteo = texto.replace(/usted/gi, 'tú').replace(/ustedes/gi, 'vosotros');

            const ustedeo = texto.replace(/tú/gi, 'usted').replace(/vosotros/gi, 'ustedes');

            const voseo = texto.replace(/tú/gi, 'vos').replace(/usted/gi, 'vos').replace(/ustedes/gi, 'vosotros');


            resultado += '<h3>Tuteo:</h3>';

            resultado += '<p>' + tuteo + '</p>';

            resultado += '<h3>Ustedeo:</h3>';

            resultado += '<p>' + ustedeo + '</p>';

            resultado += '<h3>Voseo:</h3>';

            resultado += '<p>' + voseo + '</p>';


            document.getElementById('resultado').innerHTML = resultado;

        }

    </script>
